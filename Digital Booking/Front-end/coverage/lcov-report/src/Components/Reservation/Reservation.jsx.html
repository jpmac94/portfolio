import React ,{useEffect, useState} from 'react'
import { useParams } from 'react-router-dom';
import Footer from '../Footer';
import Header from '../Header/Header';
import HeaderBlock from '../Product/HeaderBlock';
import PolicyBlock from '../Product/PolicyBlock';
 
import ContainerReservation from './ContainerReservation';
export default function Reservation() {
 
  const {id}=useParams();
 const [reservation,setReservation]=useState(null)
 useEffect(()=>{
    let requestOptions = {
      method: 'GET',
      redirect: 'follow'
    };
    
     fetch("http://localhost:8080/api/product/findById/"+id, requestOptions)
      .then(response => response.json())
      .then(result => {
        
        setReservation(result);
 
      })
      .catch(error => console.log('error', error));
  
},[id])
    return (
    <div> 
        <Header/>
        <HeaderBlock datosHeader={reservation}/>
        <ContainerReservation />
        <PolicyBlock/>
      <Footer/>
    </div>
  )
}
