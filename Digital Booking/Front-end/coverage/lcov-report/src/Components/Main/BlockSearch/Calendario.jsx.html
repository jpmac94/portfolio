import React, { useContext, useRef } from "react";
import { useState } from "react";
import DatePicker from "react-multi-date-picker";
import { BsCalendarEvent } from "react-icons/bs";
import { ContextProduct } from "../../Context/ContextProduct";
 
export default function Calendario() {
  const [value, setValue] = useState("Check in - Check out");
  const { setDatesBlockSerch } = useContext(ContextProduct);
  const datePickerRef = useRef();
 
  const aplicar = () => {
    value[0] !== undefined &&
      setDatesBlockSerch({
        startDate: value[0].year + "-" + value[0].month + "-" + value[0].day,
        endDate: value[1].year + "-" + value[1].month + "-" + value[1].day,
      });
    datePickerRef.current.closeCalendar();
  };
  let numberOfMonthsView = null;
  if (window.innerWidth < 430) {
    numberOfMonthsView = 1;
  } else {
    numberOfMonthsView = 2;
  }
 
  const weekDays = ["S", "M", "T", "W", "T", "F", "S"]
const styless={
  width: "100%"
}
  return (
    <div className="container-calendar">
      <BsCalendarEvent className="bscalendar" />
 
      <div className="c">
        <DatePicker
          weekDays={weekDays}
          placeholder="Check in - Check out"
          format="YYYY-MM-DD"
          range={true}
          numberOfMonths={numberOfMonthsView}
          minDate={new Date()}
          ref={datePickerRef}
          value={value}
          onChange={setValue}
          style={styless}
        >
          <button className="button-block-search" onClick={aplicar}>
            Aply
          </button>
        </DatePicker>
      </div>
    </div>
  );
}
